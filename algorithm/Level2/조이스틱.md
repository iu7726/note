# 조이스틱

## 제출한 풀이

```javascript
function solution(name) {
    let answer = 0;
    let basic = [...name].map(() => "A");
    let a = 65;
    let z = 90;
    let cursor = 0;
    let target = [...name].map(i => i.charCodeAt())
    let length = target.length - 1;
    console.log(target)

    while(basic.join('') != name){
        let type1 = target[cursor] - a;
        let type2 = z - target[cursor];
        let short = (type1 > type2 ? type2 : type1); //위아래 최소값
        if(type2 == 0) short++; //z인경우 예외처리
        

        basic[cursor] = name[cursor]
        
        console.log("short1",short)
        
        let next = 0;
        let prev = 0;

        let search = true;
        let order = 1;
        while(search && order <= length){
            
            let next = cursor + order;
            if(next > length) next = length
            if(target[next] != 65 && basic[next].charCodeAt() != target[next]){
                cursor = next;
                search = false;
            }
            
            let prev = cursor - order;
            if(prev < 0){
                prev = length
            }
            if(target[prev] != 65 && basic[prev].charCodeAt() != target[prev]){
                cursor = prev;
                search = false;
            }
            
            if(!search){
                 //console.log(next, target[next], prev, target[prev], cursor, order)
                short += order
            }
            
            order++;
            
        }
        
        answer += short
        console.log("last",short)
        console.log("cursor",cursor)
        console.log("ret",basic.join(''))
    }
    return answer;
}
```
